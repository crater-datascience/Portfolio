Open Targets Association Plot

This code fetches the top 15 disease associations for GPR84 from the OpenTargets database (https://www.opentargets.org/) and visualizes them.

In[1]:
import requests
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt
import numpy as np

# The new GPR84 Ensembl ID
gpr84_id = "ENSG00000139572"

url = "https://api.platform.opentargets.org/api/v4/graphql"
query = """
query gpr84Associations($ensemblId: String!) {
  target(ensemblId: $ensemblId) {
    approvedSymbol
    associatedDiseases(page: {index: 0, size: 15}) {
      rows {
        disease { name }
        score
      }
    }
  }
}
"""

variables = {"ensemblId": gpr84_id}
r = requests.post(url, json={'query': query, 'variables': variables})
data = r.json()['data']['target']['associatedDiseases']['rows']

df = pd.DataFrame([{'Disease': r['disease']['name'], 'Score': r['score']} for r in data])

# Horizontal Bar Chart Visualization
plt.figure(figsize=(10, 7))
sns.barplot(
    data=df, 
    x='Score', 
    y='Disease', 
    hue='Disease',  
    palette='viridis', 
    legend=False
)
plt.title(f"Top 15 Disease Associations for GPR84 ({gpr84_id})")
plt.xlabel("Association Score (Aggregated Evidence)")
plt.tight_layout()
plt.show()

# Define palette and extract colors appropriate for the number of tissues
num_tissues = len(df_gtex['tissueSiteDetailId'].unique())
colors = sns.color_palette("viridis", n_colors=num_tissues)

plt.figure(figsize=(22, 10))
sns.set_style("white")

# Calculate medians for the stem heights
tissue_order = df_gtex['tissueSiteDetailId'].unique()
medians = df_gtex.groupby('tissueSiteDetailId')['median'].median().reindex(tissue_order)

# 2. Draw the 'Stems' using the color list
# Each line matches its corresponding violin color
for i, (tissue, color) in enumerate(zip(tissue_order, colors)):
    plt.vlines(x=i, ymin=0.1, ymax=medians[tissue], 
               color=color, alpha=0.4, linewidth=1.5, zorder=1)

# 3. Draw the Violins with the same palette
sns.violinplot(
    data=df_gtex, 
    x='tissueSiteDetailId', 
    y='median', 
    order=tissue_order,
    palette=colors,
    hue='tissueSiteDetailId',
    legend=False,
    density_norm='width', 
    width=0.8, 
    bw_adjust=0.6, 
    cut=0, 
    inner=None, 
    zorder=2
)

# 4. Add the 'Anchor' dots
plt.scatter(x=range(len(tissue_order)), y=medians, 
            color='white', edgecolor='black', s=45, zorder=3, alpha=0.8)

# Final Polish
plt.yscale('log')
plt.ylim(0.1, 1000)
plt.xticks(rotation=90, fontsize=12)
plt.title("GPR84 Expression Profile: GTEx V10 Tissue Specificity Map", fontsize=20, pad=20)
plt.ylabel("TPM (Log Scale)", fontsize=14)
plt.xlabel("Tissue Site", fontsize=14)
sns.despine()
plt.tight_layout()
plt.show()
